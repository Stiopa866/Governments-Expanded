estate_nobles_farmers_wife = {
	max_days_active = 150
	selection_weight = {
		factor = 1
		modifier = {
			factor = 1000
			NOT = {
				total_development = 100
			}
		}
	}
	can_select = {
		NOT = {
			total_development = 100
		}
	}
	pre_effect = {
		country_event = {
			id = GE_incidents.1
		}
	}
	provinces_to_highlight = {
	}
	fail_if = {
		always = no
	}
    task_requirements = {
		OR = {
			has_country_flag = GE_incidents_1_stole_wife
			has_country_flag = GE_incidents_1_returned_wife
			has_country_flag = GE_incidents_1_killed_wife
		}
    }
    task_completed_effect = {
		custom_tooltip = GE_incidents.1.TT
		if = {
			limit = {
				has_country_flag = GE_incidents_1_stole_wife
			}
			define_consort = {
				country_of_origin = ROOT
				religion = ROOT
				culture = ROOT
				female = yes
			}
		}
		if = {
			limit = {
				has_country_flag = GE_incidents_1_returned_wife
			}
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = 10
			}
		}
		if = {
			limit = {
				has_country_flag = GE_incidents_1_killed_wife
			}
			add_stability = -1
		}
    }
    failing_effect = {
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -10
		}
    }
}
